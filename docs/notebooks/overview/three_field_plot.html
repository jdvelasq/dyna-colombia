from techminer2 import bibliometrix

bibliometrix.overview.three_fields_plot(
    directory='../../data/',
    left_criterion='authors',
    middle_criterion='countries',
    right_criterion='author_keywords',
    topics_length_left=10,
    topics_length_right=8,
).update_layout(
    width=700,
    height=500,
)
